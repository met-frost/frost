package main

import (
	"fmt"
	"io"
	"log"
	"net/http"
	"time"
)

func main() {
	client := http.Client{Timeout: 30 * time.Second}

	url := "https://frost-rc.met.no/api/v1/obs/ranked/get?incobs=false&time=latest&elementids=air_temperature&stationids=18700"
	req, err := http.NewRequest(http.MethodGet, url, nil)
	if err != nil {
		log.Fatalln("http.NewRequest() failed: ", err)
	}
	// basic auth
	req.SetBasicAuth("CLIENT_ID", "CLIENT_SECRET")

	resp, err := client.Do(req)
	if err != nil {
		log.Fatalln("client.Do() failed: ", err)
	}
	// defer closing the body
	defer resp.Body.Close()

	// parse the body of the response
	b, err := io.ReadAll(resp.Body)
	if err != nil {
		log.Fatalln("io.ReadAll() failed: ", err)
	}
	fmt.Printf("Status: %d\n", resp.StatusCode)
	fmt.Printf("Body: %s\n", string(b))
}
