import requests
import base64

url = 'https://frost-rc.met.no/api/v1/obs/ranked/get?incobs=true&time=latest&elementids=air_temperature&stationids=18700'
id_secret = "CLIENT_ID:CLIENT_SECRET"
# encode id and password as base64 string
string_bytes = id_secret.encode("ascii")
base64_bytes = base64.b64encode(string_bytes)
base64_string = base64_bytes.decode("ascii")
# format into header
header = {"Authorization" : "Basic "+ base64_string}
# send the response
response = requests.get(url, headers=header)
print(response.status_code)
if response.status_code == 200:
    print(response.json())
