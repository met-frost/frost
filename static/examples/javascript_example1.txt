function OnClick () {
    var req = new XMLHttpRequest();
    var url = "https://frost-rc.met.no/api/v1/obs/ranked/get?incobs=false&time=latest&elementids=air_temperature&stationids=18700"
    // authentication
    // NOTE: not recommended to put credentials in a clientside script, should be handled in a secure way
    var basic = "CLIENT_ID:CLIENT_SECRET";
    var hash = btoa(basic);
    // expect a json response
    req.responseType = 'json';
    // actually perform the get request
    req.open('GET', url, true);
    req.setRequestHeader("Authorization", "Basic " + hash);
    req.onload  = function() {
        var jsonResponse = req.response;
        // do something with jsonResponse
        console.log(jsonResponse)
    };
    req.send();
}
