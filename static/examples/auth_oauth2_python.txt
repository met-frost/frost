import requests
import json
import datetime

def call_frost_api_v1_oauth(client_id: str, client_secret: str):

    header = { "client_id":client_id,
               "client_secret":client_secret,
               "audience":"ODA",
               "grant_type": "urn:ietf:params:oauth:grant-type:uma-ticket"}
    response = requests.post("https://login.met.no/auth/realms/External/protocol/openid-connect/token", data=header)
    r = json.loads(response.content)
    headers = {"Authorization": "Bearer " + r["access_token"]}

    endpoint = 'https://frost-rc.met.no/api/v1/obs/base/get?'
    current_utc = datetime.datetime.utcnow()
    one_hour = datetime.timedelta(hours=1)
    one_hour_ago = current_utc - one_hour

    parameters = {
                'stationids': 76920,
                'time': (one_hour_ago).strftime('%Y-%m-%dT%TZ')+'/'+(current_utc).strftime('%Y-%m-%dT%TZ'),
                'levels': 'all',
                'incobs': 'true',
                'sensors': 'all',
                }

    frostr = requests.get(endpoint, parameters, headers=headers)
    return frostr
