library(jsonlite)

# Insert your own client ID here
client_id ='CLIENT_ID'

# Define endpoint and request parameters
endpoint <- paste0("https://", client_id, "@frost-rc.met.no/api/v1/obs/ranked/get")
stationids <- '18700,90450' #Which meteorological stations to look for (internal MET Norway numbers)
elementids <- 'wind_*' #Which elements to get. Here: All elements starting with wind, using * as wildcard
incobs <- 'true' #Set to true to get observations as well as metadata
time_interval <- '2022-12-01T00:00:00Z/2022-12-01T05:00:00Z'
# Build the URL
url <- paste0(
    endpoint, "?",
    "stationids=", stationids,
    "&time=", time_interval,
    "&elementids=", elementids,
    "&incobs=", incobs
)

#Print the request URL
print(url)

# Issue an HTTP GET request and extract JSON data
xs <- try(fromJSON(URLencode(url),flatten=T))

# Check if the request worked
if (class(xs) != 'try-error') { #if success, process response

    print("Data retrieved from frost-rc.met.no")

    #The json response is a complicated structure in R
    #We want to demonstrate how to make our own structure (as a dataframe)

    #Allocate list in order to save the data we want to concatenate at the end
    datalist = vector("list", length = length(xs$data$tseries$header.id.stationid))

    #Loop over the stationids in the response
    for( j in 1:length(xs$data$tseries$header.id.stationid)) {

	 #make a new dataframe by adding a column filled with stationids at the start of the observations dataframe
	 df_new = cbind(station = xs$data$tseries$header.id.stationid[[j]], xs$data$tseries$observations[[j]])

	 #Add a column filled with elementids to the new dataframe (position the column after number column 2)
	 df_new = data.frame(append(df_new, c(elementid=xs$data$tseries$header.extra.element.id[[j]]), after=2))

	 #Add a column filled with the measurement unit at the end of the new dataframe
	 df_new = data.frame(append(df_new, c(unit=xs$data$tseries$header.extra.element.unit[[j]])))

	 #Insert the new dataframe into our allocated list
	 datalist[[j]] <- df_new

    }

    #Concatenate the dataframes in the list into one big dataframe
    all_data = do.call(rbind, datalist)

    # Convert the time values to something R understands
    all_data$time <- as.POSIXct(all_data$time, format="%Y-%m-%dT%H:%M:%SZ", tz="UTC")

    #Make column names prettier - just because we can!
    colnames(all_data)[4] <- 'quality'
    colnames(all_data)[5] <- 'value'

    #Print the result
    print(all_data)


} else { #If error, print error message
    print("Error: the data retrieval was not successful!")
}
